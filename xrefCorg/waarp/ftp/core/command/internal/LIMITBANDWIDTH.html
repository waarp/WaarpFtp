18  package org.waarp.ftp.core.command.internal;
19  
20  import org.waarp.common.command.ReplyCode;
21  import org.waarp.common.command.exception.Reply500Exception;
22  import org.waarp.common.command.exception.Reply501Exception;
23  import org.waarp.ftp.core.command.AbstractCommand;
24  
25  /**
26   * Internal limit bandwidth command that will change the global limit bandwidth
27   * 
28   * @author Frederic Bregier
29   * 
30   */
31  public class LIMITBANDWIDTH extends AbstractCommand {
32      /*
33       * (non-Javadoc)
34       * @see org.waarp.ftp.core.command.AbstractCommand#exec()
35       */
36      public void exec() throws Reply501Exception, Reply500Exception {
37          if (!getSession().getAuth().isAdmin()) {
38              // not admin
39              throw new Reply500Exception("Command Not Allowed");
40          }
41          if (!hasArg()) {
42              // reset to default
43              getConfiguration().changeNetworkLimit(0, 0);
44              getSession().setReplyCode(ReplyCode.REPLY_200_COMMAND_OKAY,
45                      "Limit reset to default");
46              return;
47          }
48          String[] limits = getArgs();
49          long writeLimit = 0;
50          long readLimit = 0;
51          try {
52              if (limits.length == 1) {
53                  writeLimit = Long.parseLong(limits[0]);
54                  readLimit = writeLimit;
55              } else {
56                  writeLimit = Long.parseLong(limits[0]);
57                  readLimit = Long.parseLong(limits[1]);
58              }
59          } catch (NumberFormatException e) {
60              throw new Reply501Exception(getCommand() +
61                      " ([write and read limits in b/s] | [write limit in b/s] [read limit in b/s]");
62          }
63          getConfiguration().changeNetworkLimit(writeLimit, readLimit);
64          getSession().setReplyCode(ReplyCode.REPLY_200_COMMAND_OKAY,
65                  "Limit set to new values");
66      }

68  }
