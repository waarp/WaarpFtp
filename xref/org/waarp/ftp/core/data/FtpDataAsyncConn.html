115 	/**
116 	 * Constructor for Active session by default
117 	 * 
118 	 * @param session
119 	 */
120 	public FtpDataAsyncConn(FtpSession session) {
121 		this.session = session;
122 		dataChannel = null;
123 		remoteAddress = FtpChannelUtils.getRemoteInetSocketAddress(this.session
124 				.getControlChannel());
125 		remotePort = remoteAddress.getPort();
126 		setDefaultLocalPort();
127 		localAddress = new InetSocketAddress(FtpChannelUtils
128 				.getLocalInetAddress(this.session.getControlChannel()),
129 				localPort);
130 		passiveMode = false;
131 		isBind = false;
132 		transferControl = new FtpTransferControl(session);
133 	}

340 	/**
341 	 * Unbind passive connection when close the Data Channel (from channelClosed())
342 	 * 
343 	 */
344 	public void unbindPassive() {
345 		if (isBind && passiveMode) {
346 			isBind = false;
347 			InetSocketAddress local = getLocalAddress();
348 			if (dataChannel != null && dataChannel.isConnected()) {
349 				WaarpSslUtility.closingSslChannel(dataChannel);
350 			}
351 			session.getConfiguration().getFtpInternalConfiguration()
352 					.unbindPassive(local);
353 			// Previous mode was Passive so remove the current configuration if
354 			// any
355 			InetAddress remote = remoteAddress.getAddress();
356 			session.getConfiguration().delFtpSession(remote, local);
357 		}
358 		dataChannel = null;
359 		dataNetworkHandler = null;
360 	}

362 	/**
363 	 * Initialize the socket from Server side (only used in Passive)
364 	 * 
365 	 * @return True if OK
366 	 * @throws Reply425Exception
367 	 */
368 	public boolean initPassiveConnection() throws Reply425Exception {
369 		unbindPassive();
370 		if (passiveMode) {
371 			// Connection is enable but the client will do the real connection
372 			session.getConfiguration().getFtpInternalConfiguration()
373 					.bindPassive(getLocalAddress(), session.isDataSsl());
374 			isBind = true;
375 			return true;
376 		}
377 		// Connection is already prepared
378 		return true;
379 	}

425 	/**
426 	 * @return The current status in String of the different parameters
427 	 */
428 	public String getStatus() {
429 		StringBuilder builder = new StringBuilder("Data connection: ");
430 		builder.append((isConnected() ? "connected " : "not connected "));
431 		builder.append((isBind() ? "bind " : "not bind "));
432 		builder.append((isPassiveMode() ? "passive mode" : "active mode"));
433 		builder.append('\n');
434 		builder.append("Mode: ");
435 		builder.append(transferMode.name());
436 		builder.append('\n');
437 		builder.append("Structure: ");
438 		builder.append(transferStructure.name());
439 		builder.append('\n');
440 		builder.append("Type: ");
441 		builder.append(transferType.name());
442 		builder.append(' ');
443 		builder.append(transferSubType.name());
444 		return builder.toString();
445 	}

471 	public void setNewOpenedDataChannel(Channel dataChannel)

473 		this.dataChannel = dataChannel;
474 		if (dataChannel == null) {
475 			String curmode = null;
476 			if (isPassiveMode()) {
477 				curmode = "passive";
478 			} else {
479 				curmode = "active";

484 		}
