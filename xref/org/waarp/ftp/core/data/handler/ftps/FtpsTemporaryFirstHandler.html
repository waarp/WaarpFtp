29  import io.netty.channel.Channel;
30  import io.netty.channel.ChannelDuplexHandler;
31  import io.netty.channel.ChannelHandlerContext;
32  import io.netty.handler.ssl.SslHandler;
33  import io.netty.util.concurrent.Future;
34  import io.netty.util.concurrent.GenericFutureListener;

36  /**
37   * @author "Frederic Bregier"
38   *
39   */
40  class FtpsTemporaryFirstHandler extends ChannelDuplexHandler {
41      /**
42       * Internal Logger
43       */
44      private static final WaarpLogger logger = WaarpLoggerFactory
45              .getLogger(FtpsTemporaryFirstHandler.class);
46      final FtpConfiguration configuration;
47      final boolean active;
48      /**
49       * Internal store for the SessionInterface
50       */
51      FtpSession session = null;
52  
53      public FtpsTemporaryFirstHandler(FtpConfiguration configuration, boolean active) {
54          this.configuration = configuration;
55          this.active = active;
56      }
57  
58      @Override
59      public void channelRegistered(ChannelHandlerContext ctx) throws Exception {
60          ctx.channel().config().setAutoRead(false);
61          super.channelRegistered(ctx);
62      }
63  
64      protected void setSession(Channel channel) {
65          // First get the ftpSession from inetaddresses
66          for (int i = 0; i < FtpInternalConfiguration.RETRYNB; i++) {
67              session = configuration.getFtpSessionNoRemove(channel, active);
68              if (session == null) {
69                  logger.warn("Session not found at try " + i);
70                  try {
71                      Thread.sleep(FtpInternalConfiguration.RETRYINMS);
72                  } catch (InterruptedException e1) {
73                      break;
74                  }
75              } else {
76                  break;
77              }
78          }
79          if (session == null) {
80              // Not found !!!
81              logger.error("Session not found!");
82              WaarpSslUtility.closingSslChannel(channel);
83              // Problem: control connection could not be directly informed!!!
84              // Only timeout will occur
85              return;
86          }
87      }
88  
89      private void superChannelActive(ChannelHandlerContext ctx) throws Exception {
90          super.channelActive(ctx);
91      }
92  
93      @Override
94      public void channelActive(final ChannelHandlerContext ctx) throws Exception {
95          // Get the SslHandler in the current pipeline.
96          Channel channel = ctx.channel();
97  
98          if (session == null) {
99              setSession(channel);
100         }
101         if (session == null) {
102             logger.error("Cannot find session for SSL");
103             return;
104         }
105         // Server: no renegotiation still, but possible clientAuthent
106         // Mode is always as SSL Server mode.
107         SslHandler sslHandler =
108                 FtpsInitializer.waarpSslContextFactory.initInitializer(true,
109                         FtpsInitializer.waarpSslContextFactory.needClientAuthentication(),
110                         FtpChannelUtils.getRemoteInetSocketAddress(session.getControlChannel()).getAddress()
111                                 .getHostAddress(),
112                         FtpChannelUtils.getRemoteInetSocketAddress(session.getControlChannel()).getPort());
113         WaarpSslUtility.addSslOpenedChannel(channel);
114         // Get the SslHandler and begin handshake ASAP.
115         logger.debug("SSL found but need handshake: " + ctx.channel().toString());
116         final FtpsTemporaryFirstHandler myself = this;
117         WaarpSslUtility.addSslHandler(null, ctx.pipeline(), sslHandler,
118                 new GenericFutureListener<Future<? super Channel>>() {
119                     public void operationComplete(Future<? super Channel> future) throws Exception {
120                         try {
121                             logger.debug("Handshake: " + future.isSuccess() + ":" + ((Channel) future.get()).toString(),
122                                     future.cause());
123                             if (future.isSuccess()) {
124                                 logger.debug("End of initialization of SSL and data channel");
125                                 myself.superChannelActive(ctx);
126                                 ctx.pipeline().remove(myself);
127                             } else {
128                                 ctx.close();
129                             }
130                         } catch (Exception e) {
131                             ctx.close();
132                         }
133                     }
134                 });
135     }
