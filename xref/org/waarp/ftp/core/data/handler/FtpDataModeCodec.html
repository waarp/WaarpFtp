47  	/*
48  	 * 3.4.1. STREAM MODE The data is transmitted as a stream of bytes. There is no restriction on
49  	 * the representation type used; record structures are allowed. In a record structured file EOR
50  	 * and EOF will each be indicated by a two-byte control code. The first byte of the control code
51  	 * will be all ones, the escape character. The second byte will have the low order bit on and
52  	 * zeros elsewhere for EOR and the second low order bit on for EOF; that is, the byte will have
53  	 * value 1 for EOR and value 2 for EOF. EOR and EOF may be indicated together on the last byte
54  	 * transmitted by turning both low order bits on (i.e., the value 3). If a byte of all ones was
55  	 * intended to be sent as data, it should be repeated in the second byte of the control code. If
56  	 * the structure is a file structure, the EOF is indicated by the sending host closing the data
57  	 * connection and all bytes are data bytes. 3.4.2. BLOCK MODE The file is transmitted as a
58  	 * series of data blocks preceded by one or more header bytes. The header bytes contain a count
59  	 * field, and descriptor code. The count field indicates the total length of the data block in
60  	 * bytes, thus marking the beginning of the next data block (there are no filler bits). The
61  	 * descriptor code defines: last block in the file (EOF) last block in the record (EOR), restart
62  	 * marker (see the Section on Error Recovery and Restart) or suspect data (i.e., the data being
63  	 * transferred is suspected of errors and is not reliable). This last code is NOT intended for
64  	 * error control within FTP. It is motivated by the desire of sites exchanging certain types of
65  	 * data (e.g., seismic or weather data) to send and receive all the data despite local errors
66  	 * (such as "magnetic tape read errors"), but to indicate in the transmission that certain
67  	 * portions are suspect). Record structures are allowed in this mode, and any representation
68  	 * type may be used. The header consists of the three bytes. Of the 24 bits of header
69  	 * information, the 16 low order bits shall represent byte count, and the 8 high order bits
70  	 * shall represent descriptor codes as shown below. Block Header
71  	 * +----------------+----------------+----------------+ | Descriptor | Byte Count | | 8 bits |
72  	 * 16 bits | +----------------+----------------+----------------+ The descriptor codes are
73  	 * indicated by bit flags in the descriptor byte. Four codes have been assigned, where each code
74  	 * number is the decimal value of the corresponding bit in the byte. Code Meaning 128 End of
75  	 * data block is EOR 64 End of data block is EOF 32 Suspected errors in data block 16 Data block
76  	 * is a restart marker With this encoding, more than one descriptor coded condition may exist
77  	 * for a particular block. As many bits as necessary may be flagged. The restart marker is
78  	 * embedded in the data stream as an integral number of 8-bit bytes representing printable
79  	 * characters in the language being used over the control connection (e.g., default--NVT-ASCII).
80  	 * <SP> (Space, in the appropriate language) must not be used WITHIN a restart marker. For
81  	 * example, to transmit a six-character marker, the following would be sent:
82  	 * +--------+--------+--------+ |Descrptr| Byte count | |code= 16| = 6 |
83  	 * +--------+--------+--------+ +--------+--------+--------+ | Marker | Marker | Marker | | 8
84  	 * bits | 8 bits | 8 bits | +--------+--------+--------+ +--------+--------+--------+ | Marker |
85  	 * Marker | Marker | | 8 bits | 8 bits | 8 bits | +--------+--------+--------+
86  	 */

137 	protected Object decodeRecordStandard(ChannelBuffer buf, int length) {
138 		ChannelBuffer newbuf = ChannelBuffers.dynamicBuffer(length);
139 		if (lastbyte == 0xFF) {
140 			int nextbyte = buf.readByte();
141 			if (nextbyte == 0xFF) {
142 				newbuf.writeByte((byte) (lastbyte & 0xFF));
143 			} else {
144 				if (nextbyte == 1) {
145 					dataBlock.setEOR(true);
146 				} else if (nextbyte == 2) {
147 					dataBlock.setEOF(true);
148 				} else if (nextbyte == 3) {
149 					dataBlock.setEOR(true);
150 					dataBlock.setEOF(true);
151 				}
152 				lastbyte = 0;
153 			}
154 		}
155 		try {
156 			while (true) {
157 				lastbyte = buf.readByte();
158 				if (lastbyte == 0xFF) {
159 					int nextbyte = buf.readByte();
160 					if (nextbyte == 0xFF) {
161 						newbuf.writeByte((byte) (lastbyte & 0xFF));
162 					} else {
163 						if (nextbyte == 1) {
164 							dataBlock.setEOR(true);
165 						} else if (nextbyte == 2) {
166 							dataBlock.setEOF(true);
167 						} else if (nextbyte == 3) {
168 							dataBlock.setEOR(true);
169 							dataBlock.setEOF(true);
170 						}
171 					}
172 				} else {
173 					newbuf.writeByte((byte) (lastbyte & 0xFF));
174 				}
175 				lastbyte = 0;
176 			}
177 		} catch (IndexOutOfBoundsException e) {
178 			// End of read
179 		}
180 		dataBlock.setBlock(newbuf);
181 		return dataBlock;
182 	}

184 	protected Object decodeRecord(ChannelBuffer buf, int length) {
185 		FtpSeekAheadData sad = null;
186 		try {
187 			sad = new FtpSeekAheadData(buf);
188 		} catch (SeekAheadNoBackArrayException e1) {
189 			return decodeRecordStandard(buf, length);
190 		}
191 		ChannelBuffer newbuf = ChannelBuffers.dynamicBuffer(length);
192 		if (lastbyte == 0xFF) {
193 			int nextbyte = sad.bytes[sad.pos++];
194 			if (nextbyte == 0xFF) {
195 				newbuf.writeByte((byte) (lastbyte & 0xFF));
196 			} else {
197 				if (nextbyte == 1) {
198 					dataBlock.setEOR(true);
199 				} else if (nextbyte == 2) {
200 					dataBlock.setEOF(true);
201 				} else if (nextbyte == 3) {
202 					dataBlock.setEOR(true);
203 					dataBlock.setEOF(true);
204 				}
205 				lastbyte = 0;
206 			}
207 		}
208 		try {
209 			while (sad.pos < sad.limit) {
210 				lastbyte = sad.bytes[sad.pos++];
211 				if (lastbyte == 0xFF) {
212 					int nextbyte = sad.bytes[sad.pos++];
213 					if (nextbyte == 0xFF) {
214 						newbuf.writeByte((byte) (lastbyte & 0xFF));
215 					} else {
216 						if (nextbyte == 1) {
217 							dataBlock.setEOR(true);
218 						} else if (nextbyte == 2) {
219 							dataBlock.setEOF(true);
220 						} else if (nextbyte == 3) {
221 							dataBlock.setEOR(true);
222 							dataBlock.setEOF(true);
223 						}
224 					}
225 				} else {
226 					newbuf.writeByte((byte) (lastbyte & 0xFF));
227 				}
228 				lastbyte = 0;
229 			}
230 		} catch (IndexOutOfBoundsException e) {
231 			// End of read
232 		}
233 		sad.setReadPosition(0);
234 		dataBlock.setBlock(newbuf);
235 		return dataBlock;

239 	 * (non-Javadoc)
240 	 * @see org.jboss.netty.handler.codec.frame.FrameDecoder#decode(org.jboss.netty
241 	 * .channel.ChannelHandlerContext, org.jboss.netty.channel.Channel,
242 	 * org.jboss.netty.buffer.ChannelBuffer)
243 	 */
244 	@Override
245 	protected Object decode(ChannelHandlerContext ctx, Channel channel,
246 			ChannelBuffer buf) throws Exception {
247 		// First test if the connection is fully ready (block might be
248 		// transfered
249 		// by client before connection is ready)
250 		if (!isReady) {
251 			codecLocked.await();
252 			isReady = true;
253 		}
254 		// If STREAM Mode, no task to do, just next filter
255 		if (mode == TransferMode.STREAM) {
256 			dataBlock = new DataBlock();
257 			int length = buf.readableBytes();
258 			// Except if RECORD Structure!
259 			if (structure == TransferStructure.RECORD) {
260 				return decodeRecord(buf, length);
261 			}
262 
263 			dataBlock.setBlock(buf.readBytes(length));
264 			return dataBlock;
265 		} else if (mode == TransferMode.BLOCK) {
266 			// Now we are in BLOCK Mode
267 			// Make sure if the length field was received.
268 			if (buf.readableBytes() < 3) {
269 				// The length field was not received yet - return null.
270 				// This method will be invoked again when more packets are
271 				// received and appended to the buffer.
272 				return null;
273 			}
274 
275 			// The length field is in the buffer.
276 
277 			// Mark the current buffer position before reading the length field
278 			// because the whole frame might not be in the buffer yet.
279 			// We will reset the buffer position to the marked position if
280 			// there's not enough bytes in the buffer.
281 			buf.markReaderIndex();
282 
283 			if (dataBlock == null) {
284 				dataBlock = new DataBlock();
285 			}
286 			// Read the descriptor
287 			dataBlock.setDescriptor(buf.readByte());
288 
289 			// Read the length field.
290 			byte upper = buf.readByte();
291 			byte lower = buf.readByte();
292 			dataBlock.setByteCount(upper, lower);
293 
294 			// Make sure if there's enough bytes in the buffer.
295 			if (buf.readableBytes() < dataBlock.getByteCount()) {
296 				// The whole bytes were not received yet - return null.
297 				// This method will be invoked again when more packets are
298 				// received and appended to the buffer.
299 
300 				// Reset to the marked position to read the length field again
301 				// next time.
302 				buf.resetReaderIndex();
303 
304 				return null;
305 			}
306 			if (dataBlock.getByteCount() > 0) {
307 				// There's enough bytes in the buffer. Read it.
308 				dataBlock.setBlock(buf.readBytes(dataBlock.getByteCount()));
309 			}
310 			DataBlock returnDataBlock = dataBlock;
311 			// Free the datablock for next frame
312 			dataBlock = null;
313 			// Successfully decoded a frame. Return the decoded frame.
314 			return returnDataBlock;
315 		}
316 		// Type unimplemented
317 		throw new InvalidArgumentException("Mode unimplemented: " + mode.name());
318 	}

320 	protected ChannelBuffer encodeRecordStandard(DataBlock msg, ChannelBuffer buffer) {
321 		ChannelBuffer newbuf = ChannelBuffers.dynamicBuffer(msg
322 				.getByteCount());
323 		int newbyte = 0;
324 		try {
325 			while (true) {
326 				newbyte = buffer.readByte();
327 				if (newbyte == 0xFF) {
328 					newbuf.writeByte((byte) (newbyte & 0xFF));
329 				}
330 				newbuf.writeByte((byte) (newbyte & 0xFF));
331 			}
332 		} catch (IndexOutOfBoundsException e) {
333 			// end of read
334 		}
335 		int value = 0;
336 		if (msg.isEOF()) {
337 			value += 2;
338 		}
339 		if (msg.isEOR()) {
340 			value += 1;
341 		}
342 		if (value > 0) {
343 			newbuf.writeByte((byte) 0xFF);
344 			newbuf.writeByte((byte) (value & 0xFF));
345 		}
346 		msg.clear();
347 		return newbuf;
348 	}

387 	/**
388 	 * Encode a DataBlock in the correct format for Mode
389 	 * 
390 	 * @param msg
391 	 * @return the ChannelBuffer or null when the last block is already done
392 	 * @throws InvalidArgumentException
393 	 */
394 	protected ChannelBuffer encode(DataBlock msg)
395 			throws InvalidArgumentException {
396 		if (msg.isCleared()) {
397 			return null;
398 		}
399 		ChannelBuffer buffer = msg.getBlock();
400 		if (mode == TransferMode.STREAM) {
401 			// If record structure, special attention
402 			if (structure == TransferStructure.RECORD) {
403 				return encodeRecord(msg, buffer);
404 			}
405 			msg.clear();
406 			return buffer;
407 		} else if (mode == TransferMode.BLOCK) {
408 			int length = msg.getByteCount();
409 			ChannelBuffer newbuf = ChannelBuffers
410 					.dynamicBuffer(length > 0xFFFF ? 0xFFFF + 3 : length + 3);
411 			byte[] header = new byte[3];
412 			// Is there any data left
413 			if (length == 0) {
414 				// It could be an empty block for EOR or EOF
415 				if (msg.isEOF() || msg.isEOR()) {
416 					header[0] = msg.getDescriptor();
417 					header[1] = 0;
418 					header[2] = 0;
419 					newbuf.writeBytes(header);
420 					// Next call will be the last one
421 					msg.clear();
422 					// return the last block
423 					return newbuf;
424 				}
425 				// This was the very last call
426 				msg.clear();
427 				// return the end of encode
428 				return null;
429 			}
430 			// Is this a Restart so only Markers
431 			if (msg.isRESTART()) {
432 				header[0] = msg.getDescriptor();
433 				header[1] = msg.getByteCountUpper();
434 				header[2] = msg.getByteCountLower();
435 				newbuf.writeBytes(header);
436 				newbuf.writeBytes(msg.getByteMarkers());
437 				// Next call will be the last one
438 				msg.clear();
439 				// return the last block
440 				return newbuf;
441 			}
442 			// Work on sub block, ignoring descriptor since it is not the last
443 			// one
444 			if (length > 0xFFFF) {
445 				header[0] = 0;
446 				header[1] = (byte) 0xFF;
447 				header[2] = (byte) 0xFF;
448 				newbuf.writeBytes(header);
449 				// Now take the first 0xFFFF bytes from buffer
450 				newbuf.writeBytes(msg.getBlock(), 0xFFFF);
451 				length -= 0xFFFF;
452 				msg.setByteCount(length);
453 				// return the sub block
454 				return newbuf;
455 			}
456 			// Last final block, using the descriptor
457 			header[0] = msg.getDescriptor();
458 			header[1] = msg.getByteCountUpper();
459 			header[2] = msg.getByteCountLower();
460 			newbuf.writeBytes(header);
461 			// real data
462 			newbuf.writeBytes(buffer, length);
463 			// Next call will be the last one
464 			msg.clear();
465 			// return the last block
466 			return newbuf;
467 		}
468 		// Mode unimplemented
469 		throw new InvalidArgumentException("Mode unimplemented: " + mode.name());
470 	}

511 	/**
512 	 * Coder part, taking a DataBlock and converting it to ChannelBuffer
513 	 * 
514 	 * @param ctx
515 	 * @param evt
516 	 * @throws Exception
517 	 */
518 	private void writeRequested(ChannelHandlerContext ctx, MessageEvent evt)
519 			throws Exception {
520 		if (!(evt.getMessage() instanceof DataBlock)) {
521 			Channels.write(ctx, evt.getFuture(), evt.getMessage());
522 			return;
523 			// since SSL in beginning will send a handshake => take care
524 			/*throw new InvalidArgumentException("Incorrect write object: " +
525 					evt.getMessage().getClass().getName());*/
526 		}
527 		// First test if the connection is fully ready (block might be
528 		// transfered
529 		// by client before connection is ready)
530 		if (!isReady) {
531 			codecLocked.await();
532 			isReady = true;
533 		}
534 		DataBlock newDataBlock = (DataBlock) evt.getMessage();
535 		ChannelBuffer next = encode(newDataBlock);
536 		// Could be splitten in several block
537 		while (next != null) {
538 			Channels.write(ctx, evt.getFuture(), next);
539 			next = encode(newDataBlock);
540 		}
