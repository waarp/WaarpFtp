337 	@Override
338 	public String toString() {
339 		String mesg = "FtpSession: ";
340 		if (ftpAuth != null) {
341 			mesg += "User: " + ftpAuth.getUser() + "/" + ftpAuth.getAccount() + " ";
342 		}
343 		if (currentCommand != null) {
344 			mesg += "CMD: " + currentCommand.getCommand() + " " +
345 					currentCommand.getArg() + " ";
346 		}
347 		if (replyCode != null) {
348 			mesg += "Reply: " + (answer != null ? answer : replyCode.getMesg()) +
349 					" ";
350 		}
351 		if (dataConn != null) {
352 			mesg += dataConn.toString();
353 		}
354 		if (ftpDir != null) {
355 			try {
356 				mesg += " PWD: " + ftpDir.getPwd();
357 			} catch (CommandAbstractException e) {
358 			}
359 		}
360 		return mesg + "\n";
361 	}
