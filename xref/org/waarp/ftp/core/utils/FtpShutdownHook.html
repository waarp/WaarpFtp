21  package org.waarp.ftp.core.utils;
22  
23  import org.waarp.common.utility.WaarpShutdownHook;
24  import org.waarp.ftp.core.config.FtpConfiguration;
25  
26  /**
27   * @author "Frederic Bregier"
28   *
29   */
30  public class FtpShutdownHook extends WaarpShutdownHook {
31  
32  	protected FtpConfiguration configuration;
33  	/**
34  	 * @param configuration
35  	 */
36  	public FtpShutdownHook(ShutdownConfiguration configuration, FtpConfiguration ftpconfiguration) {
37  		super(configuration);
38  		this.configuration = ftpconfiguration;
39  	}
40  
41  	@Override
42  	protected void exit() {
43  		FtpChannelUtils.exit(this.configuration);
44  		try {
45  			Thread.sleep(this.configuration.TIMEOUTCON/2);
46  		} catch (InterruptedException e) {
47  		}
48  	}
49  
50  }
