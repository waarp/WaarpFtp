18  package org.waarp.ftp.core.command.parameter;
19  
20  import java.net.InetAddress;
21  import java.net.InetSocketAddress;
22  
23  import org.waarp.common.command.ReplyCode;
24  import org.waarp.common.command.exception.Reply501Exception;
25  import org.waarp.common.logging.WaarpLogger;
26  import org.waarp.common.logging.WaarpLoggerFactory;
27  import org.waarp.ftp.core.command.AbstractCommand;
28  import org.waarp.ftp.core.utils.FtpChannelUtils;
29  
30  /**
31   * PORT command
32   * 
33   * @author Frederic Bregier
34   * 
35   */
36  public class PORT extends AbstractCommand {
37      /**
38       * Internal Logger
39       */
40      private static final WaarpLogger logger = WaarpLoggerFactory
41              .getLogger(PORT.class);
42  
43      /*
44       * (non-Javadoc)
45       * @see org.waarp.ftp.core.command.AbstractCommand#exec()
46       */
47      public void exec() throws Reply501Exception {
48          // First Check if any argument
49          if (!hasArg()) {
50              InetSocketAddress inetSocketAddress = getSession().getDataConn()
51                      .getRemoteAddress();
52              getSession().getDataConn().setActive(inetSocketAddress);
53              getSession().setReplyCode(
54                      ReplyCode.REPLY_200_COMMAND_OKAY,
55                      "PORT command successful on (" +
56                              inetSocketAddress.toString() + ")");
57              return;
58          }
59          // Check if Inet Address is OK
60          InetSocketAddress inetSocketAddress = FtpChannelUtils
61                  .getInetSocketAddress(getArg());
62          if (inetSocketAddress == null) {
63              // ERROR
64              throw new Reply501Exception("Need correct Inet Address as argument");
65          }
66          // Check if the Client address is the same as given
67          InetAddress remoteAddress = inetSocketAddress.getAddress();
68          InetAddress trueRemoteAddress = getSession().getDataConn()
69                  .getRemoteAddress().getAddress();
70          if (!remoteAddress.equals(trueRemoteAddress)) {
71              // ERROR
72              logger.warn("Given Inet Address {} mismatchs actual client Address {}", remoteAddress, trueRemoteAddress);
73              throw new Reply501Exception(
74                      "Given Inet Address mismatchs actual client Address");
75          }
76          // OK now try to initialize connection (not open)
77          getSession().getDataConn().setActive(inetSocketAddress);
78          getSession().setReplyCode(
79                  ReplyCode.REPLY_200_COMMAND_OKAY,
80                  "PORT command successful on (" + inetSocketAddress.toString() +
81                          ")");
82      }
83  }
