18  package org.waarp.ftp.core.command.parameter;
19  
20  import org.waarp.common.command.ReplyCode;
21  import org.waarp.common.command.exception.Reply501Exception;
22  import org.waarp.common.command.exception.Reply504Exception;
23  import org.waarp.common.exception.InvalidArgumentException;
24  import org.waarp.ftp.core.command.AbstractCommand;
25  import org.waarp.ftp.core.command.FtpArgumentCode;
26  import org.waarp.ftp.core.command.FtpArgumentCode.TransferSubType;
27  
28  /**
29   * TYPE command
30   * 
31   * @author Frederic Bregier
32   * 
33   */
34  public class TYPE extends AbstractCommand {
35      @Override
36      public void exec() throws Reply501Exception, Reply504Exception {
37          // First Check if any argument
38          if (!hasArg()) {
39              getSession().getDataConn().setType(
40                      FtpArgumentCode.TransferType.ASCII);
41              getSession().getDataConn().setSubType(TransferSubType.NONPRINT);
42              getSession().setReplyCode(
43                      ReplyCode.REPLY_200_COMMAND_OKAY,
44                      "Type set to " + FtpArgumentCode.TransferType.ASCII.name() +
45                              " " + TransferSubType.NONPRINT);
46              return;
47          }
48          FtpArgumentCode.TransferType transferType;
49          String[] types = getArgs();
50          try {
51              transferType = FtpArgumentCode.getTransferType(types[0].charAt(0));
52          } catch (InvalidArgumentException e) {
53              throw new Reply501Exception("Unrecognize Type: " + getArg());
54          }
55          if (transferType == FtpArgumentCode.TransferType.ASCII) {
56              getSession().getDataConn().setType(transferType);
57          } else if (transferType == FtpArgumentCode.TransferType.IMAGE) {
58              getSession().getDataConn().setType(transferType);
59          } else {
60              throw new Reply504Exception("Type not implemented: " +
61                      transferType.name());
62          }
63          // Look at the subtype or format control
64          if (types.length > 2) {
65              TransferSubType transferSubType = null;
66              for (int i = 1; i < types.length; i++) {
67                  if (types[i].length() != 0) {
68                      try {
69                          transferSubType = FtpArgumentCode
70                                  .getTransferSubType(types[i].charAt(0));
71                      } catch (InvalidArgumentException e) {
72                          throw new Reply501Exception(
73                                  "Unrecognize Format Control: " + types[i]);
74                      }
75                      if (transferSubType != TransferSubType.NONPRINT) {
76                          throw new Reply504Exception(
77                                  "Format Control not implemented: " +
78                                          transferSubType.name());
79                      }
80                  }
81              }
82              getSession().getDataConn().setSubType(TransferSubType.NONPRINT);
83          } else {
84              getSession().getDataConn().setSubType(TransferSubType.NONPRINT);
85          }
86          getSession().setReplyCode(
87                  ReplyCode.REPLY_200_COMMAND_OKAY,
88                  "Type set to " + transferType.name() + " " +
89                          TransferSubType.NONPRINT);
90      }

92  }
