18  package org.waarp.ftp.core.command.service;
19  
20  import java.util.List;
21  
22  import org.waarp.common.command.exception.CommandAbstractException;
23  import org.waarp.ftp.core.command.AbstractCommand;
24  
25  /**
26   * NLST command
27   * 
28   * @author Frederic Bregier
29   * 
30   */
31  public class NLST extends AbstractCommand {
32  
33  	/*
34  	 * (non-Javadoc)
35  	 * @see org.waarp.ftp.core.command.AbstractCommand#exec()
36  	 */
37  	public void exec() throws CommandAbstractException {
38  		String path = null;
39  		List<String> files = null;
40  		if (!hasArg()) {
41  			path = getSession().getDir().getPwd();
42  			files = getSession().getDir().list(path);
43  		} else {
44  			path = getArg();
45  			if (path.startsWith("-l") || path.startsWith("-L")) {
46  				// This should be a LIST command
47  				String[] paths = getArgs();
48  				if (paths.length > 1) {
49  					files = getSession().getDir().listFull(paths[1], true);
50  				} else {
51  					files = getSession().getDir().listFull(
52  							getSession().getDir().getPwd(), true);
53  				}
54  			} else {
55  				files = getSession().getDir().list(path);
56  			}
57  		}
58  		getSession().openDataConnection();
59  		getSession().getDataConn().getFtpTransferControl().setNewFtpTransfer(
60  				getCode(), files, path);
61  	}
62  
63  }
