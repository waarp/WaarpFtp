18  package org.waarp.ftp.core.command.service;
19  
20  import org.waarp.common.command.ReplyCode;
21  import org.waarp.common.command.exception.CommandAbstractException;
22  import org.waarp.common.command.exception.Reply450Exception;
23  import org.waarp.common.command.exception.Reply501Exception;
24  import org.waarp.common.command.exception.Reply550Exception;
25  import org.waarp.ftp.core.command.AbstractCommand;
26  import org.waarp.ftp.core.file.FtpFile;
27  
28  /**
29   * DELE command
30   * 
31   * @author Frederic Bregier
32   * 
33   */
34  public class DELE extends AbstractCommand {
35  
36  	/*
37  	 * (non-Javadoc)
38  	 * @see org.waarp.ftp.core.command.AbstractCommand#exec()
39  	 */
40  	public void exec() throws CommandAbstractException {
41  		if (!hasArg()) {
42  			invalidCurrentCommand();
43  			throw new Reply501Exception("Need a pathname as argument");
44  		}
45  		String filename = getArg();
46  		FtpFile file = getSession().getDir().setFile(filename, false);
47  		if (file != null) {
48  			if (file.delete()) {
49  				getSession().setReplyCode(
50  						ReplyCode.REPLY_250_REQUESTED_FILE_ACTION_OKAY,
51  						"\"" + file.getFile() + "\" FtpFile is deleted");
52  				return;
53  			}
54  			throw new Reply450Exception("Delete operation not allowed");
55  		}
56  		throw new Reply550Exception("Filename not allowed");
57  	}
58  
59  }
