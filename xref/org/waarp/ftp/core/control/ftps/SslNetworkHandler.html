21  package org.waarp.ftp.core.control.ftps;
22  
23  import org.jboss.netty.channel.Channel;
24  import org.jboss.netty.channel.ChannelFuture;
25  import org.jboss.netty.channel.ChannelFutureListener;
26  import org.jboss.netty.channel.ChannelHandlerContext;
27  import org.jboss.netty.channel.ChannelStateEvent;
28  import org.jboss.netty.handler.ssl.SslHandler;
29  import org.waarp.common.logging.WaarpInternalLogger;
30  import org.waarp.common.logging.WaarpInternalLoggerFactory;
31  import org.waarp.ftp.core.control.NetworkHandler;
32  import org.waarp.ftp.core.session.FtpSession;
33  
34  /**
35   * @author "Frederic Bregier"
36   *
37   */
38  public class SslNetworkHandler extends NetworkHandler {
39  	/**
40  	 * Internal Logger
41  	 */
42  	private static final WaarpInternalLogger logger = WaarpInternalLoggerFactory
43  			.getLogger(SslNetworkHandler.class);
44  
45  	/**
46  	 * @param session
47  	 */
48  	public SslNetworkHandler(FtpSession session) {
49  		super(session);
50  	}
51  
52  	/**
53  	 * Remover from SSL HashMap
54  	 */
55  	private static final ChannelFutureListener remover = new ChannelFutureListener() {
56  		public void operationComplete(
57  				ChannelFuture future) {
58  			logger.debug("SSL finishing: "+future.getChannel().getId());
59  		}
60  	};
61  
62  	/**
63  	 * Add the Channel as SSL handshake is over
64  	 * 
65  	 * @param channel
66  	 */
67  	private static void addSslConnectedChannel(Channel channel) {
68  		channel.getCloseFuture().addListener(remover);
69  	}
70  
71  
72  	@Override
73  	public void channelOpen(ChannelHandlerContext ctx, ChannelStateEvent e)
74  			throws Exception {
75  		Channel channel = e.getChannel();
76  		logger.debug("Add channel to ssl " +channel.getId());
77  		addSslConnectedChannel(channel);
78  		super.channelOpen(ctx, e);
79  	}
80  
81  	/**
82  	 * To be extended to inform of an error to SNMP support
83  	 * @param error1
84  	 * @param error2
85  	 */
86  	protected void callForSnmp(String error1, String error2) {
87  		// ignore
88  	}
89  
90  	@Override
91  	public void channelConnected(ChannelHandlerContext ctx, ChannelStateEvent e) {
92  		// Get the SslHandler in the current pipeline.
93  		// We added it in NetworkSslServerPipelineFactory.
94  		final SslHandler sslHandler = ctx.getPipeline().get(SslHandler.class);
95  		if (sslHandler != null) {
96  			// Get the SslHandler and begin handshake ASAP.
97  			// Get notified when SSL handshake is done.
98  			ChannelFuture handshakeFuture;
99  			handshakeFuture = sslHandler.handshake();
100 			try {
101 				handshakeFuture.await();
102 			} catch (InterruptedException e1) {
103 			}
104 			logger.debug("Handshake: " + handshakeFuture.isSuccess(), handshakeFuture.getCause());
105 			if (!handshakeFuture.isSuccess()) {
106 				String error2 = handshakeFuture.getCause() != null ?
107 						handshakeFuture.getCause().getMessage() : "During Handshake";
108 				callForSnmp("SSL Connection Error", error2);
109 				handshakeFuture.getChannel().close();
110 				return;
111 			}
112 		} else {
113 			logger.error("SSL Not found");
114 		}
115 		super.channelConnected(ctx, e);
116 	}
