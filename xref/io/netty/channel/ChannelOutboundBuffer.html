107     /**
108      * Add given message to this {@link ChannelOutboundBuffer}. The given {@link ChannelPromise} will be notified once
109      * the message was written.
110      */
111     public void addMessage(Object msg, int size, ChannelPromise promise) {
112         Entry entry = Entry.newInstance(msg, size, total(msg), promise);
113         if (tailEntry == null) {
114             flushedEntry = null;
115             tailEntry = entry;
116         } else {
117             Entry tail = tailEntry;
118             tail.next = entry;
119             tailEntry = entry;
120         }
121         if (unflushedEntry == null) {
122             unflushedEntry = entry;
123         }
124 
125         // increment pending bytes after adding message to the unflushed arrays.
126         // See https://github.com/netty/netty/issues/1619
127         incrementPendingOutboundBytes(size);
128     }

130     /**
131      * Add a flush to this {@link ChannelOutboundBuffer}. This means all previous added messages are marked as flushed
132      * and so you will be able to handle them.
133      */
134     public void addFlush() {
135         // There is no need to process all entries if there was already a flush before and no new messages
136         // where added in the meantime.
137         //
138         // See https://github.com/netty/netty/issues/2577
139         Entry entry = unflushedEntry;
140         if (entry != null) {
141             if (flushedEntry == null) {
142                 // there is no flushedEntry yet, so start with the entry
143                 flushedEntry = entry;
144             }
145             do {
146                 flushed ++;
147                 if (!entry.promise.setUncancellable()) {
148                     // Was cancelled so make sure we free up memory and notify about the freed bytes
149                     int pending = entry.cancel();
150                     decrementPendingOutboundBytes(pending);
151                 }
152                 entry = entry.next;
153             } while (entry != null);
154 
155             // All flushed so reset unflushedEntry
156             unflushedEntry = null;
157         }
158     }

160     /**
161      * Increment the pending bytes which will be written at some point.
162      * This method is thread-safe!
163      */
164     void incrementPendingOutboundBytes(long size) {
165         if (size == 0) {
166             return;
167         }
168 
169         long newWriteBufferSize = TOTAL_PENDING_SIZE_UPDATER.addAndGet(this, size);
170         if (newWriteBufferSize >= channel.config().getWriteBufferHighWaterMark()) {
171             setUnwritable();
172         }
173     }
174 
175     /**
176      * Decrement the pending bytes which will be written at some point.
177      * This method is thread-safe!
178      */
179     void decrementPendingOutboundBytes(long size) {
180         if (size == 0) {
181             return;
182         }
183 
184         long newWriteBufferSize = TOTAL_PENDING_SIZE_UPDATER.addAndGet(this, -size);
185         if (newWriteBufferSize == 0 || newWriteBufferSize <= channel.config().getWriteBufferLowWaterMark()) {
186             setWritable();
187         }
188     }

190     private static long total(Object msg) {
191         if (msg instanceof ByteBuf) {
192             return ((ByteBuf) msg).readableBytes();
193         }
194         if (msg instanceof FileRegion) {
195             return ((FileRegion) msg).count();
196         }
197         if (msg instanceof ByteBufHolder) {
198             return ((ByteBufHolder) msg).content().readableBytes();
199         }
200         return -1;
201     }

229     /**
230      * Will remove the current message, mark its {@link ChannelPromise} as success and return {@code true}. If no
231      * flushed message exists at the time this method is called it will return {@code false} to signal that no more
232      * messages are ready to be handled.
233      */
234     public boolean remove() {
235         Entry e = flushedEntry;
236         if (e == null) {
237             return false;
238         }
239         Object msg = e.msg;
240 
241         ChannelPromise promise = e.promise;
242         int size = e.pendingSize;
243 
244         removeEntry(e);
245 
246         if (!e.cancelled) {
247             // only release message, notify and decrement if it was not canceled before.
248             ReferenceCountUtil.safeRelease(msg);
249             safeSuccess(promise);
250             decrementPendingOutboundBytes(size);
251         }
252 
253         // recycle the entry
254         e.recycle();
255 
256         return true;
257     }
258 
259     /**
260      * Will remove the current message, mark its {@link ChannelPromise} as failure using the given {@link Throwable}
261      * and return {@code true}. If no   flushed message exists at the time this method is called it will return
262      * {@code false} to signal that no more messages are ready to be handled.
263      */
264     public boolean remove(Throwable cause) {
265         Entry e = flushedEntry;
266         if (e == null) {
267             return false;
268         }
269         Object msg = e.msg;
270 
271         ChannelPromise promise = e.promise;
272         int size = e.pendingSize;
273 
274         removeEntry(e);
275 
276         if (!e.cancelled) {
277             // only release message, fail and decrement if it was not canceled before.
278             ReferenceCountUtil.safeRelease(msg);
279 
280             safeFail(promise, cause);
281             decrementPendingOutboundBytes(size);
282         }
283 
284         // recycle the entry
285         e.recycle();
286 
287         return true;
288     }

290     private void removeEntry(Entry e) {
291         if (-- flushed == 0) {
292             // processed everything
293             flushedEntry = null;
294             if (e == tailEntry) {
295                 tailEntry = null;
296                 unflushedEntry = null;
297             }
298         } else {
299             flushedEntry = e.next;
300         }
301     }

303     /**
304      * Removes the fully written entries and update the reader index of the partially written entry.
305      * This operation assumes all messages in this buffer is {@link ByteBuf}.
306      */
307     public void removeBytes(long writtenBytes) {
308         for (;;) {
309             Object msg = current();
310             if (!(msg instanceof ByteBuf)) {
311                 assert writtenBytes == 0;
312                 break;
313             }
314 
315             final ByteBuf buf = (ByteBuf) msg;
316             final int readerIndex = buf.readerIndex();
317             final int readableBytes = buf.writerIndex() - readerIndex;
318 
319             if (readableBytes <= writtenBytes) {
320                 if (writtenBytes != 0) {
321                     progress(readableBytes);
322                     writtenBytes -= readableBytes;
323                 }
324                 remove();
325             } else { // readableBytes > writtenBytes
326                 if (writtenBytes != 0) {
327                     buf.readerIndex(readerIndex + (int) writtenBytes);
328                     progress(writtenBytes);
329                 }
330                 break;
331             }
332         }
333     }

335     /**
336      * Returns an array of direct NIO buffers if the currently pending messages are made of {@link ByteBuf} only.
337      * {@link #nioBufferCount()} and {@link #nioBufferSize()} will return the number of NIO buffers in the returned
338      * array and the total number of readable bytes of the NIO buffers respectively.
339      * <p>
340      * Note that the returned array is reused and thus should not escape
341      * {@link AbstractChannel#doWrite(ChannelOutboundBuffer)}.
342      * Refer to {@link NioSocketChannel#doWrite(ChannelOutboundBuffer)} for an example.
343      * </p>
344      */
345     public ByteBuffer[] nioBuffers() {
346         long nioBufferSize = 0;
347         int nioBufferCount = 0;
348         final InternalThreadLocalMap threadLocalMap = InternalThreadLocalMap.get();
349         ByteBuffer[] nioBuffers = NIO_BUFFERS.get(threadLocalMap);
350         Entry entry = flushedEntry;
351         while (isFlushedEntry(entry) && entry.msg instanceof ByteBuf) {
352             if (!entry.cancelled) {
353                 ByteBuf buf = (ByteBuf) entry.msg;
354                 final int readerIndex = buf.readerIndex();
355                 final int readableBytes = buf.writerIndex() - readerIndex;
356 
357                 if (readableBytes > 0) {
358                     nioBufferSize += readableBytes;
359                     int count = entry.count;
360                     if (count == -1) {
361                         //noinspection ConstantValueVariableUse
362                         entry.count = count =  buf.nioBufferCount();
363                     }
364                     int neededSpace = nioBufferCount + count;
365                     if (neededSpace > nioBuffers.length) {
366                         nioBuffers = expandNioBufferArray(nioBuffers, neededSpace, nioBufferCount);
367                         NIO_BUFFERS.set(threadLocalMap, nioBuffers);
368                     }
369                     if (count == 1) {
370                         ByteBuffer nioBuf = entry.buf;
371                         if (nioBuf == null) {
372                             // cache ByteBuffer as it may need to create a new ByteBuffer instance if its a
373                             // derived buffer
374                             entry.buf = nioBuf = buf.internalNioBuffer(readerIndex, readableBytes);
375                         }
376                         nioBuffers[nioBufferCount ++] = nioBuf;
377                     } else {
378                         ByteBuffer[] nioBufs = entry.bufs;
379                         if (nioBufs == null) {
380                             // cached ByteBuffers as they may be expensive to create in terms
381                             // of Object allocation
382                             entry.bufs = nioBufs = buf.nioBuffers();
383                         }
384                         nioBufferCount = fillBufferArray(nioBufs, nioBuffers, nioBufferCount);
385                     }
386                 }
387             }
388             entry = entry.next;
389         }
390         this.nioBufferCount = nioBufferCount;
391         this.nioBufferSize = nioBufferSize;
392 
393         return nioBuffers;
394     }

396     private static int fillBufferArray(ByteBuffer[] nioBufs, ByteBuffer[] nioBuffers, int nioBufferCount) {
397         for (ByteBuffer nioBuf: nioBufs) {
398             if (nioBuf == null) {
399                 break;
400             }
401             nioBuffers[nioBufferCount ++] = nioBuf;
402         }
403         return nioBufferCount;
404     }

406     private static ByteBuffer[] expandNioBufferArray(ByteBuffer[] array, int neededSpace, int size) {
407         int newCapacity = array.length;
408         do {
409             // double capacity until it is big enough
410             // See https://github.com/netty/netty/issues/1890
411             newCapacity <<= 1;
412 
413             if (newCapacity < 0) {
414                 throw new IllegalStateException();
415             }
416 
417         } while (neededSpace > newCapacity);
418 
419         ByteBuffer[] newArray = new ByteBuffer[newCapacity];
420         System.arraycopy(array, 0, newArray, 0, size);
421 
422         return newArray;
423     }

472     private void setUserDefinedWritability(int index) {
473         final int mask = ~writabilityMask(index);
474         for (;;) {
475             final int oldValue = unwritable;
476             final int newValue = oldValue & mask;
477             if (UNWRITABLE_UPDATER.compareAndSet(this, oldValue, newValue)) {
478                 if (oldValue != 0 && newValue == 0) {
479                     channel.pipeline().fireChannelWritabilityChanged();
480                 }
481                 break;
482             }
483         }
484     }
485 
486     private void clearUserDefinedWritability(int index) {
487         final int mask = writabilityMask(index);
488         for (;;) {
489             final int oldValue = unwritable;
490             final int newValue = oldValue | mask;
491             if (UNWRITABLE_UPDATER.compareAndSet(this, oldValue, newValue)) {
492                 if (oldValue == 0 && newValue != 0) {
493                     channel.pipeline().fireChannelWritabilityChanged();
494                 }
495                 break;
496             }
497         }
498     }

500     private static int writabilityMask(int index) {
501         if (index < 1 || index > 31) {
502             throw new IllegalArgumentException("index: " + index + " (expected: 1~31)");
503         }
504         return 1 << index;
505     }

548     void failFlushed(Throwable cause) {
549         // Make sure that this method does not reenter.  A listener added to the current promise can be notified by the
550         // current thread in the tryFailure() call of the loop below, and the listener can trigger another fail() call
551         // indirectly (usually by closing the channel.)
552         //
553         // See https://github.com/netty/netty/issues/1501
554         if (inFail) {
555             return;
556         }
557 
558         try {
559             inFail = true;
560             for (;;) {
561                 if (!remove(cause)) {
562                     break;
563                 }
564             }
565         } finally {
566             inFail = false;
567         }
568     }

570     void close(final ClosedChannelException cause) {
571         if (inFail) {
572             channel.eventLoop().execute(new Runnable() {
573                 @Override
574                 public void run() {
575                     close(cause);
576                 }
577             });
578             return;
579         }
580 
581         inFail = true;
582 
583         if (channel.isOpen()) {
584             throw new IllegalStateException("close() must be invoked after the channel is closed.");
585         }
586 
587         if (!isEmpty()) {
588             throw new IllegalStateException("close() must be invoked after all flushed writes are handled.");
589         }
590 
591         // Release all unflushed messages.
592         try {
593             Entry e = unflushedEntry;
594             while (e != null) {
595                 // Just decrease; do not trigger any events via decrementPendingOutboundBytes()
596                 int size = e.pendingSize;
597                 TOTAL_PENDING_SIZE_UPDATER.addAndGet(this, -size);
598 
599                 if (!e.cancelled) {
600                     ReferenceCountUtil.safeRelease(e.msg);
601                     safeFail(e.promise, cause);
602                 }
603                 e = e.recycleAndGetNext();
604             }
605         } finally {
606             inFail = false;
607         }
608     }

610     private static void safeSuccess(ChannelPromise promise) {
611         if (!(promise instanceof VoidChannelPromise) && !promise.trySuccess()) {
612             logger.warn("Failed to mark a promise as success because it is done already: {}", promise);

616     private static void safeFail(ChannelPromise promise, Throwable cause) {

631     /**
632      * Call {@link MessageProcessor#processMessage(Object)} for each flushed message
633      * in this {@link ChannelOutboundBuffer} until {@link MessageProcessor#processMessage(Object)}
634      * returns {@code false} or there are no more flushed messages to process.
635      */
636     public void forEachFlushedMessage(MessageProcessor processor) throws Exception {
637         if (processor == null) {
638             throw new NullPointerException("processor");
639         }
640 
641         Entry entry = flushedEntry;
642         if (entry == null) {
643             return;
644         }
645 
646         do {
647             if (!entry.cancelled) {
648                 if (!processor.processMessage(entry.msg)) {
649                     return;
650                 }
651             }
652             entry = entry.next;
653         } while (isFlushedEntry(entry));
654     }

656     private boolean isFlushedEntry(Entry e) {
657         return e != null && e != unflushedEntry;
658     }

669         private static final Recycler<Entry> RECYCLER = new Recycler<Entry>() {
670             @Override
671             protected Entry newObject(Handle handle) {
672                 return new Entry(handle);
673             }
674         };

676         private final Handle handle;
677         Entry next;
678         Object msg;
679         ByteBuffer[] bufs;
680         ByteBuffer buf;
681         ChannelPromise promise;
682         long progress;
683         long total;
684         int pendingSize;
685         int count = -1;
686         boolean cancelled;
687 
688         private Entry(Handle handle) {
689             this.handle = handle;
690         }
691 
692         static Entry newInstance(Object msg, int size, long total, ChannelPromise promise) {
693             Entry entry = RECYCLER.get();
694             entry.msg = msg;
695             entry.pendingSize = size;
696             entry.total = total;
697             entry.promise = promise;
698             return entry;
699         }
700 
701         int cancel() {
702             if (!cancelled) {
703                 cancelled = true;
704                 int pSize = pendingSize;
705 
706                 // release message and replace with an empty buffer
707                 ReferenceCountUtil.safeRelease(msg);
708                 msg = Unpooled.EMPTY_BUFFER;
709 
710                 pendingSize = 0;
711                 total = 0;
712                 progress = 0;
713                 bufs = null;
714                 buf = null;
715                 return pSize;
716             }
717             return 0;
718         }
719 
720         void recycle() {
721             next = null;
722             bufs = null;
723             buf = null;
724             msg = null;
725             promise = null;
726             progress = 0;
727             total = 0;
728             pendingSize = 0;
729             count = -1;
730             cancelled = false;
731             RECYCLER.recycle(this, handle);
732         }
733 
734         Entry recycleAndGetNext() {
735             Entry next = this.next;
736             recycle();
737             return next;
738         }
