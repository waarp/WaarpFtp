32  /**
33   * <p>This implementation of the {@link AbstractTrafficShapingHandler} is for global
34   * traffic shaping, that is to say a global limitation of the bandwidth, whatever
35   * the number of opened channels.</p>
36   * <p>Note the index used in <code>OutboundBuffer.setUserDefinedWritability(index, boolean)</code> is <b>2</b>.</p>
37   *
38   * <p>The general use should be as follow:</p>
39   * <ul>
40   * <li><p>Create your unique GlobalTrafficShapingHandler like:</p>
41   * <p><tt>GlobalTrafficShapingHandler myHandler = new GlobalTrafficShapingHandler(executor);</tt></p>
42   * <p>The executor could be the underlying IO worker pool</p>
43   * <p><tt>pipeline.addLast(myHandler);</tt></p>
44   *
45   * <p><b>Note that this handler has a Pipeline Coverage of "all" which means only one such handler must be created
46   * and shared among all channels as the counter must be shared among all channels.</b></p>
47   *
48   * <p>Other arguments can be passed like write or read limitation (in bytes/s where 0 means no limitation)
49   * or the check interval (in millisecond) that represents the delay between two computations of the
50   * bandwidth and so the call back of the doAccounting method (0 means no accounting at all).</p>
51   *
52   * <p>A value of 0 means no accounting for checkInterval. If you need traffic shaping but no such accounting,
53   * it is recommended to set a positive value, even if it is high since the precision of the
54   * Traffic Shaping depends on the period where the traffic is computed. The highest the interval,
55   * the less precise the traffic shaping will be. It is suggested as higher value something close
56   * to 5 or 10 minutes.</p>
57   *
58   * <p>maxTimeToWait, by default set to 15s, allows to specify an upper bound of time shaping.</p>
59   * </li>
60   * <li>In your handler, you should consider to use the <code>channel.isWritable()</code> and
61   * <code>channelWritabilityChanged(ctx)</code> to handle writability, or through
62   * <code>future.addListener(new GenericFutureListener())</code> on the future returned by
63   * <code>ctx.write()</code>.</li>
64   * <li>You shall also consider to have object size in read or write operations relatively adapted to
65   * the bandwidth you required: for instance having 10 MB objects for 10KB/s will lead to burst effect,
66   * while having 100 KB objects for 1 MB/s should be smoothly handle by this TrafficShaping handler.</li>
67   * <li><p>Some configuration methods will be taken as best effort, meaning
68   * that all already scheduled traffics will not be
69   * changed, but only applied to new traffics.</p>
70   * <p>So the expected usage of those methods are to be used not too often,
71   * accordingly to the traffic shaping configuration.</p></li>
72   * </ul>
73   *
74   * Be sure to call {@link #release()} once this handler is not needed anymore to release all internal resources.
75   * This will not shutdown the {@link EventExecutor} as it may be shared, so you need to do this by your own.
76   */

233     private PerChannel getOrSetPerChannel(ChannelHandlerContext ctx) {
234         // ensure creation is limited to one thread per channel
235         Channel channel = ctx.channel();
236         Integer key = channel.hashCode();
237         PerChannel perChannel = channelQueues.get(key);
238         if (perChannel == null) {
239             perChannel = new PerChannel();
240             perChannel.messagesQueue = new ArrayDeque<ToSend>();
241             perChannel.queueSize = 0L;
242             perChannel.lastReadTimestamp = TrafficCounter.milliSecondFromNano();
243             perChannel.lastWriteTimestamp = perChannel.lastReadTimestamp;
244             channelQueues.put(key, perChannel);
245         }
246         return perChannel;
247     }

255     @Override
256     public void handlerRemoved(ChannelHandlerContext ctx) throws Exception {
257         Channel channel = ctx.channel();
258         Integer key = channel.hashCode();
259         PerChannel perChannel = channelQueues.remove(key);
260         if (perChannel != null) {
261             // write operations need synchronization
262             synchronized (perChannel) {
263                 if (channel.isActive()) {
264                     for (ToSend toSend : perChannel.messagesQueue) {
265                         long size = calculateSize(toSend.toSend);
266                         trafficCounter.bytesRealWriteFlowControl(size);
267                         perChannel.queueSize -= size;
268                         queuesSize.addAndGet(-size);
269                         ctx.write(toSend.toSend, toSend.promise);
270                     }
271                 } else {
272                     queuesSize.addAndGet(-perChannel.queueSize);
273                     for (ToSend toSend : perChannel.messagesQueue) {
274                         if (toSend.toSend instanceof ByteBuf) {
275                             ((ByteBuf) toSend.toSend).release();
276                         }
277                     }
278                 }
279                 perChannel.messagesQueue.clear();
280             }
281         }
282         releaseWriteSuspended(ctx);
283         releaseReadSuspended(ctx);
284         super.handlerRemoved(ctx);
285     }

287     @Override
288     long checkWaitReadTime(final ChannelHandlerContext ctx, long wait, final long now) {
289         Integer key = ctx.channel().hashCode();
290         PerChannel perChannel = channelQueues.get(key);
291         if (perChannel != null) {
292             if (wait > maxTime && now + wait - perChannel.lastReadTimestamp > maxTime) {
293                 wait = maxTime;
294             }
295         }
296         return wait;
297     }

308     private static final class ToSend {

322     @Override
323     void submitWrite(final ChannelHandlerContext ctx, final Object msg,
324             final long size, final long writedelay, final long now,
325             final ChannelPromise promise) {
326         Channel channel = ctx.channel();
327         Integer key = channel.hashCode();
328         PerChannel perChannel = channelQueues.get(key);
329         if (perChannel == null) {
330             // in case write occurs before handlerAdded is raized for this handler
331             // imply a synchronized only if needed
332             perChannel = getOrSetPerChannel(ctx);
333         }
334         final ToSend newToSend;
335         long delay = writedelay;
336         boolean globalSizeExceeded = false;
337         // write operations need synchronization
338         synchronized (perChannel) {
339             if (writedelay == 0 && perChannel.messagesQueue.isEmpty()) {
340                 trafficCounter.bytesRealWriteFlowControl(size);
341                 ctx.write(msg, promise);
342                 perChannel.lastWriteTimestamp = now;
343                 return;
344             }
345             if (delay > maxTime && now + delay - perChannel.lastWriteTimestamp > maxTime) {
346                 delay = maxTime;
347             }
348             newToSend = new ToSend(delay + now, msg, size, promise);
349             perChannel.messagesQueue.addLast(newToSend);
350             perChannel.queueSize += size;
351             queuesSize.addAndGet(size);
352             checkWriteSuspend(ctx, delay, perChannel.queueSize);
353             if (queuesSize.get() > maxGlobalWriteSize) {
354                 globalSizeExceeded = true;
355             }
356         }
357         if (globalSizeExceeded) {
358             setUserDefinedWritability(ctx, false);
359         }
360         final long futureNow = newToSend.relativeTimeAction;
361         final PerChannel forSchedule = perChannel;
362         ctx.executor().schedule(new Runnable() {
363             @Override
364             public void run() {
365                 sendAllValid(ctx, forSchedule, futureNow);
366             }
367         }, delay, TimeUnit.MILLISECONDS);
368     }
369 
370     private void sendAllValid(final ChannelHandlerContext ctx, final PerChannel perChannel, final long now) {
371         // write operations need synchronization
372         synchronized (perChannel) {
373             ToSend newToSend = perChannel.messagesQueue.pollFirst();
374             for (; newToSend != null; newToSend = perChannel.messagesQueue.pollFirst()) {
375                 if (newToSend.relativeTimeAction <= now) {
376                     long size = newToSend.size;
377                     trafficCounter.bytesRealWriteFlowControl(size);
378                     perChannel.queueSize -= size;
379                     queuesSize.addAndGet(-size);
380                     ctx.write(newToSend.toSend, newToSend.promise);
381                     perChannel.lastWriteTimestamp = now;
382                 } else {
383                     perChannel.messagesQueue.addFirst(newToSend);
384                     break;
385                 }
386             }
387             if (perChannel.messagesQueue.isEmpty()) {
388                 releaseWriteSuspended(ctx);
389             }
390         }
391         ctx.flush();
392     }
